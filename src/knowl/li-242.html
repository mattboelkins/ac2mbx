\begin{align*} p'(z) \amp = \amp \lim_{h \to 0} \frac{p(z+h) - p(z)}{h}\\ \amp = \amp \lim_{h \to 0} \frac{(z+h)^2 - z^2}{h}\\ \amp = \amp \lim_{h \to 0} \frac{z^2 + 2zh + h^2 - z^2}{h}\\ \amp = \amp \lim_{h \to 0} \frac{2zh + h^2}{h}\\ \amp = \amp \lim_{h \to 0} 2z + h\\ \amp = \amp 2z \end{align*}
